//ts-check
import { LitElement, html, css } from "lit";

/**
 * Cart icon button. 
 */
export class CartIcon extends LitElement{
    static styles = css`
        div{
            display: inline-block;
            position: relative;
        }

        #icon{
            background: red;
            padding: 0.25em;
            border-radius: 100%; 
        }

        span{
            color: white;
            position: absolute;
            left: 50%;
            transform: translateX(-50%);
            top: 41%;
            font-family: var(--Poppins);
            
        }
    `
    /**
     * itemsInCart 
     * @type {number} 
     */
    static properties = {
        itemsInCart: {type: Number}
    }


    render(){
        return html`
        <div><iconify-icon align id='icon' icon="heroicons:shopping-bag" style="color: white; font-size: 40px;"></iconify-icon><span>${this.itemsInCart}</span></div>
        
        `
    }
}

customElements.define('cart-icon', CartIcon)
